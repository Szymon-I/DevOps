minikube start - run minikube vm
minikube service <service_name> - assign external ip to service (for load balancer)
minikube addons enable ingress - add ingress nginx controller 

kubectl get all - get all components
kubectl get pod/service/deployment - get info about k8s components
kubectl get pod -o wide - get detailed info about pods (e.g. ip address)
kubectl apply -f <filename> - execute k8s configuration from file (can also update)
kubectl get pod -n kube-system - get minikube related pods
kubectl get ns - get namespaces
kubectl rollout restart deployment <deployment name> - restart deployment
kubectl logs --follow <pod name> attach log console (infinitive)
kubectl get pods --show-labels - get pods with labels
